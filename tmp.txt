
lint:python:
  stage: lint
  image: python:3.11
  before_script:
    - pip config set global.index-url https://pypi.tuna.tsinghua.edu.cn/simple
    - python -m pip install -U pip -i https://pypi.tuna.tsinghua.edu.cn/simple --default-timeout=100
    - pip install click
    - pip install flake8 black isort
  script:
    # 后端 Python 代码检查
    - flake8 backend/app/ --count --select=E9,F63,F7,F82 --show-source --statistics || true
    - flake8 backend/app/ --count --max-complexity=10 --max-line-length=127 --statistics || true
    - black --check backend/app/ || true
    - isort --check-only backend/app/ || true
  allow_failure: true
  rules:
    - if: $CI_PIPELINE_SOURCE == "push"  # 每次 push 触发

lint:frontend:
  stage: lint
  image:
    name: node:20-bullseye
    pull_policy: if-not-present
  variables:
    GIT_STRATEGY: clone
    GIT_CLEAN_FLAGS: none # ← 禁用git clean
    FF_USE_NEW_BASH_EVAL: "true"
  before_script:
    - cd frontend
    - npm config set registry https://registry.npmmirror.com
    - rm -rf node_modules .cache dist
    - npm ci --prefer-offline --no-audit
  script:
    - npm run lint || true
  allow_failure: true
  rules:
    - if: $CI_PIPELINE_SOURCE == "push"  # 每次 push 触发


# 基础连接测试
deploy:connection-test:
  <<: *deploy_base
  script:
    - echo "Testing SSH connectivity..."
    - sshpass -p "$DEPLOY_PASSWORD" ssh -o StrictHostKeyChecking=no -o ConnectTimeout=5 "${DEPLOY_USER}@${DEPLOY_SERVER}" 'echo "Connection OK"'
  allow_failure: false

# 环境探测（非必须）
deploy:test:
  <<: *deploy_base
  script:
    - echo "Testing remote environment..."
    - |
      sshpass -p "$DEPLOY_PASSWORD" ssh \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=5 \
        $DEPLOY_USER@$DEPLOY_SERVER '
          echo "=== 环境测试 ==="
          echo "用户: $(whoami)"
          echo "主机: $(hostname)"
          echo "时间: $(date)"
          echo "目录: $(pwd)"
          ls -la /home/TimeCapsule/ || true
          sudo -n systemctl status nginx --no-pager -l | head -3 && echo "Nginx 权限 OK" || echo "Nginx 权限测试失败"
          echo "环境测试完成"
        '
  allow_failure: true
